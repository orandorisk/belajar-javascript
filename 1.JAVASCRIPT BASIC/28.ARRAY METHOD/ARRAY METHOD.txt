//KALI INI MEMBAHAS FILTER DAN FIND METHOD (UNTUK MENCARI ELEMEN PADA ARRAY)
..FILTER MENCARI ELEMEN DAN MENGEMBALIKAN BANYAK NILAI						|	
  SYNTAX SEDERHANA FILTER : (mencari nilai pada array dan mengembalikannya dalam bentuk array	|
	var angka = [1,2,10,5,20,3,6,8,4];							|	METHOD FILTER AKAN MENEMUKAN SEBUAH NILAI BERADA DI DALAM SEBUAH ARRAY
	angka2 = angka.filter(function(x){							|	BUAT VAR BARU UNTUK MENAMPUNG NILAI FILTER , ARRAY SEBELUMNYA DIPANGGIL MENGGUNAKAN FILTER
		return x > 5;									|	NILAI X YANG AKAN DICARI ADALAH ANGKA LEBIH DARI 5 MAKA 
	});											|
	console.log(angka2.join(' - '));							|	DAN PADA CONSOLE ANGKA DITAMPILKAN NILAI 7 - 6 - 8 - 9
..FIND MENCARI ELEMEN DAN HANYA DAPAT MENEMUKAN SATU NILAI DAN PENGEMBALIAN BUKAN DALAM BENTUK ARRAY
  SYNTAX SEDERHANA FIND : (find tidak dapat menggunakan method join karena hanya 1 nilai saja)
	var angka = [1,2,10,5,20,3,6,8,4];								|	METHOD FIND AKAN MENEMUKAN SEBUAH NILAI BERADA DI DALAM SEBUAH ARRAY
	angka2 = angka.find(function(x){								|	BUAT VAR BARU UNTUK MENAMPUNG FIND , ARRAY SEBELUMNYA DIPANGGIL MENGGUNAKAN FIND
		return x == 4;										|	NILAI X YANG AKAN DICARI ADALAH ANGKA 5 MAKA
	});												|
	console.log(angka2);										| 	DAN PADA CONSOLE ANGKA DITAMPILKAN NILAI 5 * INGAT FIND HANYA DAPAT MENAMPILKAN SATU NILAI JIKA DIGUNAKAN JOIN MAKA AKAN ERROR
